package com.cedarsoft.rest.generator.test.jaxb;

import javax.ws.rs.core.UriBuilder;
import com.cedarsoft.jaxb.JaxbObject;
import com.cedarsoft.rest.JaxbMapping;
import com.cedarsoft.rest.JaxbMappingContext;
import com.cedarsoft.rest.generator.test.FooModel;

public class FooModelJaxbMapping
    extends JaxbMapping<FooModel, FooModelJaxb, FooModelJaxbStub>
{


    public FooModelJaxbMapping(BarModelJaxbMapping barModelJaxbMapping) {
        getDelegatesMapping().addMapping(BarModelJaxb.class, barModelJaxbMapping);
    }

    @Override
    protected void setUris(JaxbObject object, UriBuilder uriBuilder) {
        object.setHref(uriBuilder.path("foomodel").path("{id}").build(object.getId()));
    }

    @Override
    protected FooModelJaxb createJaxbObject(FooModel object, JaxbMappingContext context) {
        FooModelJaxb jaxbObject = new FooModelJaxb();
        jaxbObject.setId(object.getId());
        jaxbObject.setSingleBar(getStub(BarModelJaxb.class, object.getSingleBar(), context));
        jaxbObject.setTheBars(getStub(BarModelJaxb.class, object.getTheBars(), context));
        return jaxbObject;
    }

    @Override
    protected FooModelJaxbStub createJaxbObjectStub(FooModel object, JaxbMappingContext context) {
        FooModelJaxbStub jaxbObject = new FooModelJaxbStub();
        jaxbObject.setId(object.getId());
        jaxbObject.setSingleBar(getStub(BarModelJaxb.class, object.getSingleBar(), context));
        return jaxbObject;
    }

}
