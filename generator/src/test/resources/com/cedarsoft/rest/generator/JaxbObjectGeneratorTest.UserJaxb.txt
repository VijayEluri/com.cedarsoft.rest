package com.cedarsoft.rest.generator.test.jaxb;

import javax.ws.rs.core.UriBuilder;
import com.cedarsoft.jaxb.JaxbObject;
import com.cedarsoft.rest.JaxbMapping;
import com.cedarsoft.rest.JaxbMappingContext;
import com.cedarsoft.rest.generator.test.User;

public class UserMapping
    extends JaxbMapping<User, UserJaxb, UserJaxbStub>
{
    @NonNls
    @NotNull
    public static final String PATH_USERS = "users";
    @NonNls
    public static final String PATH_ID = "{id}";


    public UserJaxbMapping(UserJaxbMapping userMapping) {
        getDelegatesMapping().addMapping(UserJaxb.class, userMapping);
    }

    @Override
    protected void setUris(JaxbObject object, UriBuilder uriBuilder) {
        object.setHref(uriBuilder.path(PATH_USERS).path(PATH_ID).build(object.getId()));
    }

    @Override
    protected UserJaxb createJaxbObject(User object, JaxbMappingContext context) {
        User.Jaxb jaxbObject = new User.Jaxb();
        jaxbObject.setEmail(object.getEmail());
        jaxbObject.setName(object.getName());
        jaxbObject.setFriends(getStub(UserJaxb.class, object.getFriends(), context));
        return jaxbObject;
    }

    @Override
    protected UserJaxbStub createJaxbObjectStub(User object, JaxbMappingContext context) {
        User.Stub stub = new User.Stub();
        stub.setEmail(object.getEmail());
        stub.setName(object.getName());
        return stub;
    }

}
