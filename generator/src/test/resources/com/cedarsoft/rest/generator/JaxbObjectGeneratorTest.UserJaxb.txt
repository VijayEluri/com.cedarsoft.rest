package com.cedarsoft.rest.generator.test.jaxb;

import java.net.URISyntaxException;
import javax.ws.rs.core.UriBuilder;
import com.cedarsoft.jaxb.JaxbObject;
import com.cedarsoft.rest.JaxbMapping;
import com.cedarsoft.rest.JaxbMappingContext;
import com.cedarsoft.rest.generator.test.User;

public class UserMapping
    extends JaxbMapping<User, com.cedarsoft.rest.generator.test.jaxb.User.Jaxb, com.cedarsoft.rest.generator.test.jaxb.User.Stub>
{

    public final static String PATH = "users";
    public final static String ID = "{id}";

    public UserMapping(GroupMapping groupMapping, UserMapping userMapping) {
        getDelegatesMapping().addMapping(com.cedarsoft.rest.generator.test.jaxb.Group.Jaxb.class, com.cedarsoft.rest.generator.test.jaxb.Group.Stub.class, groupMapping);
        getDelegatesMapping().addMapping(com.cedarsoft.rest.generator.test.jaxb.User.Jaxb.class, com.cedarsoft.rest.generator.test.jaxb.User.Stub.class, userMapping);
    }

    @Override
    protected void setUris(JaxbObject object, UriBuilder uriBuilder) {
        object.setHref(uriBuilder.path(PATH).path(ID).build(object.getId()));
    }

    @Override
    protected com.cedarsoft.rest.generator.test.jaxb.User.Jaxb createJaxbObject(User object, JaxbMappingContext context)
        throws URISyntaxException
    {
        com.cedarsoft.rest.generator.test.jaxb.User.Jaxb jaxbObject = new com.cedarsoft.rest.generator.test.jaxb.User.Jaxb();
        jaxbObject.setEmail(object.getEmail());
        jaxbObject.setName(object.getName());
        jaxbObject.setGroup(getStub(com.cedarsoft.rest.generator.test.jaxb.Group.Stub.class, object.getGroup(), context));
        jaxbObject.setFriends(getStub(com.cedarsoft.rest.generator.test.jaxb.User.Stub.class, object.getFriends(), context));
        return jaxbObject;
    }

    @Override
    protected com.cedarsoft.rest.generator.test.jaxb.User.Stub createJaxbObjectStub(User object, JaxbMappingContext context)
        throws URISyntaxException
    {
        com.cedarsoft.rest.generator.test.jaxb.User.Stub stub = new com.cedarsoft.rest.generator.test.jaxb.User.Stub();
        stub.setEmail(object.getEmail());
        stub.setName(object.getName());
        stub.setGroup(getStub(com.cedarsoft.rest.generator.test.jaxb.Group.Stub.class, object.getGroup(), context));
        return stub;
    }

}
